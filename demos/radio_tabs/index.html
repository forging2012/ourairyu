Tabs with radio buttons

HTML

<div class="radioTabs">
  <div class="radioGroup">
    <label><input name="bank" type="radio" value="person"> 个人</label>
    <label><input name="bank" type="radio" value="company"> 公司</label>
  </div>
  <div class="tab_bank" data-flag="person">个人贷款</div>
  <div class="tab_bank" data-flag="company">公司贷款</div>
</div>

JavaScript

$("body").on("change", ".radioTabs :radio[name='bank']", function() {
  $(".radioTabs .tab_bank.show").removeClass("show");
  $(".radioTabs .tab_bank[data-flag='" + $(this).val() + "']").addClass("show");
});
